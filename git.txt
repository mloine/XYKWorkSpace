-- 下载指定github仓库
git clone “”
-- 放入暂存区
git add ./
--本地提交
git commit -m "备注"
-- 上传到github
git push origin master

-- 从服务器拉代码
git pull

-- 显示日志
git log

--优化显示日志
git log --pretty=oneline


--回退版本到缓存区
git reset HEAD

--从缓存区回退版本到工作区域
git checkout "目标文件"

查看用户名和邮箱地址：

git config user.name

git config user.email

修改用户名和邮箱地址

git config --global user.name  "xxxx"

git config --global user.email  "xxxx"
